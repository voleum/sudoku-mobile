class DatabaseMigrator {
  private db: SQLiteDatabase;

  constructor(database: SQLiteDatabase) {
    this.db = database;
  }

  async getCurrentVersion(): Promise<string> {
    try {
      const result = await this.db.executeSql(
        'SELECT version FROM migrations WHERE applied = 1 ORDER BY applied_at DESC LIMIT 1'
      );

      if (result[0].rows.length > 0) {
        return result[0].rows.item(0).version;
      }
    } catch (error) {
      // –¢–∞–±–ª–∏—Ü–∞ migrations –Ω–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç - —ç—Ç–æ –ø–µ—Ä–≤—ã–π –∑–∞–ø—É—Å–∫
    }

    return '1.0.0';
  }

  async applyMigration(migration: Migration): Promise<void> {
    const transaction = async () => {
      // –ó–∞–ø–∏—Å—ã–≤–∞–µ–º –º–∏–≥—Ä–∞—Ü–∏—é
      await this.db.executeSql(
        `INSERT INTO migrations (version, name, up_script, down_script, description, checksum)
         VALUES (?, ?, ?, ?, ?, ?)`,
        [
          migration.version,
          migration.name,
          migration.upScript,
          migration.downScript || null,
          migration.description,
          this.calculateChecksum(migration.upScript)
        ]
      );

      // –ü—Ä–∏–º–µ–Ω—è–µ–º SQL —Å–∫—Ä–∏–ø—Ç
      const statements = migration.upScript.split(';').filter(s => s.trim());
      for (const statement of statements) {
        if (statement.trim()) {
          await this.db.executeSql(statement);
        }
      }

      // –û—Ç–º–µ—á–∞–µ–º –∫–∞–∫ –ø—Ä–∏–º–µ–Ω–µ–Ω–Ω—É—é
      await this.db.executeSql(
        'UPDATE migrations SET applied = 1, applied_at = ? WHERE version = ?',
        [Math.floor(Date.now() / 1000), migration.version]
      );
    };

    await this.db.transaction(transaction);
  }

  private calculateChecksum(script: string): string {
    // –ü—Ä–æ—Å—Ç–∞—è –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–∞—è —Å—É–º–º–∞ –¥–ª—è –≤–∞–ª–∏–¥–∞—Ü–∏–∏
    return Buffer.from(script).toString('base64').slice(0, 32);
  }
}
